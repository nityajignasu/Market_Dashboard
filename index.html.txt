function _demoQ(sym) {
  const b = DQ[sym] || {c:+(50+Math.random()*400).toFixed(2), dp:+((Math.random()-.5)*3).toFixed(2), pc:100};
  return {c:+(b.c*(1+(Math.random()-.5)*.003)).toFixed(2), dp:+(b.dp+(Math.random()-.5)*.15).toFixed(2), pc:b.pc};
}

async function getQuote(sym) {
  if (DEMO) return _demoQ(sym);
  try {
    const ctrl = new AbortController();
    const tid  = setTimeout(() => ctrl.abort(), 5000);
    const r    = await fetch(`https://finnhub.io/api/v1/quote?symbol=${sym}&token=${KEY}`, {signal:ctrl.signal});
    clearTimeout(tid);
    const d = await r.json();
    // Fallback if API returns no price (bad key, CORS from file://, rate limit)
    if (!d || !d.c || d.c === 0) return _demoQ(sym);
    return d;
  } catch {
    // Silently fall back to demo data (e.g. opened as local file)
    return _demoQ(sym);
  }
}

async function getNews() {
  if (DEMO) return demoNews();
  try {
    const r = await fetch(`https://finnhub.io/api/v1/news?category=general&token=${KEY}`);
    const d = await r.json();
    return Array.isArray(d) ? d.slice(0,8) : demoNews();
  } catch { return demoNews(); }
}

const sleep = ms => new Promise(r => setTimeout(r, ms));

async function fetchAll(syms) {
  const out = {};
  for (let i = 0; i < syms.length; i++) {
    const q = await getQuote(syms[i]);
    if (q?.c) out[syms[i]] = q;
    if (!DEMO && i < syms.length-1) await sleep(110);
  }
  return out;
}

async function renderNews() {
  const news = await getNews();
  const html = news.slice(0,6).map(n => {
    const h = n.headline || n.title || '';
    const bc = /surge|rally|gain|beat|rise|record|strong|bullish/i.test(h) ? 'tag-green'
             : /fall|drop|decline|fear|warn|miss|cut|bearish/i.test(h) ? 'tag-red' : 'tag-dim';
    const bl = /surge|rally|gain|beat|rise|record|strong|bullish/i.test(h) ? 'Bullish'
             : /fall|drop|decline|fear|warn|miss|cut|bearish/i.test(h) ? 'Bearish' : 'Neutral';
    const ago = n.datetime ? timeAgo(n.datetime) : '';
    return `<div class="news-item">
      <div class="news-headline"><a href="${n.url||'#'}" target="_blank" rel="noopener">${h||'—'}</a></div>
      <div class="news-meta"><span>${n.source||'—'}</span>${ago?`<span>${ago}</span>`:''}<span class="tag ${bc}">${bl}</span></div>
    </div>`;
  }).join('') || '<div class="empty">No news available</div>';

  ['newsBeg','newsAdv'].forEach(id => { const e=document.getElementById(id); if(e) e.innerHTML=html; });
}

function timeAgo(ts) {
  const s = Math.floor(Date.now()/1000 - ts);
  if (s < 3600) return Math.floor(s/60)+'m ago';
  if (s < 86400) return Math.floor(s/3600)+'h ago';
  return Math.floor(s/86400)+'d ago';

// ═══════════════════════════
// STRESS TESTER
// ═══════════════════════════
function calcStress() {
  const sliders = document.querySelectorAll('#stressSliders input[type=range]');
  const eq = +sliders[0].value, bond = +sliders[1].value, cash = +sliders[2].value;
  document.getElementById('eqV').textContent   = eq+'%';
  document.getElementById('bondV').textContent  = bond+'%';
  document.getElementById('cashV').textContent  = cash+'%';
  const t = (eq+bond+cash) || 100;
  const sc = [
    {label:'COVID (2020)',   eq:-34,bond:8,  cash:.5, sub:'-34% equities'},
    {label:'GFC (2008–09)', eq:-55,bond:10, cash:2,  sub:'-55% equities'},
    {label:'Dot-com (2000)',eq:-49,bond:14, cash:4,  sub:'-49% equities'},
    {label:'10% Correction',eq:-10,bond:3,  cash:.5, sub:'-10% equities'},
  ];
  document.getElementById('stressOut').innerHTML = sc.map(s => {
    const r = (eq/t*s.eq + bond/t*s.bond + cash/t*s.cash)/100;
    const c = r >= 0 ? 'var(--green)' : 'var(--red)';
    return `<div class="stress-card">
      <div class="stress-sc-label">${s.label}</div>
      <div class="stress-sc-val" style="color:${c}">${r>=0?'+':''}${(r*100).toFixed(1)}%</div>
      <div class="stress-sc-sub">${s.sub}</div>
    </div>`;
  }).join('');
}
